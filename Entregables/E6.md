## Código
> #include<Servo.h>
>
> const int pinEMG = A0;
> 
> //Declaración otras variables
> int signalEMG = 0; //Valor con el que inicia la señal emg
> int umbral = 300;
> //Declaración servomotor
> Servo servo1;
>
> void setup() {
>  //Configuración previa
> Serial.begin(115200); 
> servo1.attach(9); //pin servo
> }
>
>  //Código principal
> void loop() {
> signalEMG = analogRead(pinEMG);
>
> if(signalEMG>umbral){
> servo1.write(180);
> }
> else{
>    servo1.write(0);
>  }
> Serial.println(signalEMG);
> delay(100);
> }
